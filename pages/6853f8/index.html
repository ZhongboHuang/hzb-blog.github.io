    从零搭建 vue3 + vite2 项目

    # unplugin-auto-import/vite 的使用

    1. 下载

    pnpm install -D unplugin-auto-import
    1

    2. vite.config.ts 配置

    import AutoImport from 'unplugin-auto-import/vite'

export default defineConfig({
  plugins: [
    ...
     AutoImport ({
      include: [
        /\.[tj]sx?$/, 
        /\.vue$/, 
        /\.vue\?vue/, 
        /\.md$/,
      ],
      imports: [
        'vue', 
        'vue-router', 
        'pinia'
      ], // 自动导入 vue，vue-router，pinia 的 api
      dts: 'src/auto-import.d.ts' // 生成 `auto-import.d.ts` 全局声明
    })
  ]
})

    3. 修改 tsconfig.json 配置

    插件配置之后，运行代码时会自动在根目录下(通过dts配置可修改路径)生成一个auto-import.d.ts文件，需要将此文件添加到tsconfig.json中，否则在使用api时会提示未定义：

    {
  "include": [
    "src/auto-import.d.ts"
  ]
}

    # vite-plugin-svg-icons 的使用

    1. 安装 vite-plugin-svg-icons 插件

    pnpm install vite-plugin-svg-icons -D
    1

    2. 配置 vite.config.ts

    import { createSvgIconsPlugin } from 'vite-plugin-svg-icons'
import path from 'path'

export default defineConfig({
  ...
  plugins: [
    createSvgIconsPlugin({
      // 指定svg文件地址
      iconDirs: [path.resolve(process.cwd(), 'src/assets/svg')],
      // 指定 symbolId 格式
      symbolId: '[name]'
    })
  ]
})

    3. 配置 main.ts

    4. 封装 SvgIcon 组件

    <template>
  <svg aria-hidden="true" :class="svgClass">
    <use :href="symbolId" />
  </svg>
</template>

<script setup>
import { computed } from 'vue'

interface SvgProps {
  prefix?: string
  name: string
  className?: string
}

const props = withDefault(defineProps<SvgProps>(), {
  prefix: 'icon',
  className: ''
})

const symbolId = computed(() => `#${props.name}`)
const svgClass = computed(() => {
  if(props.className) {
    return `svg-icon ${props.className}`
  }
  return `svg-icon`
})
</script>

<style lang='less' scoped>
.svg-icon {
  vertical-align: -0.1em; /* 因icon大小被设置为和字体大小一致，而span等标签的下边缘会和字体的基线对齐，故需设置一个往下的偏移比例，来纠正视觉上的未对齐效果 */
  fill: currentColor; /* 定义元素的颜色，currentColor是一个变量，这个变量的值就表示当前元素的color值，如果当前元素未设置color值，则从父元素继承 */
  overflow: hidden;
}
</style>

    # 配置 vue3 代码片段

    文件 -> 首选项 -> 配置用户代码片段

    选择 vue.json

    "Vue3 template": {
  "prefix": "vue3",
  "body": [
    "<script setup lang='ts' name=''>",
    "</script>",
    "<template>",
    "\t<div class=''>",
    "\t</div>",
    "</template>",
    "<style lang='less' scoped>",
    "</style>"
  ],
  "description": "vue3 template"
}
